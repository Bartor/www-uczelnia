int** generate(int k) {
	int** t = malloc((1 << k)*sizeof(int*));
	for (int i = 0; i < (1 << k); i++) {
		t[i] = malloc(k * sizeof(int));
		int ones = numberOfSetBits(i);
		int limit = (ones + 1 > k - ones ? 1 << (ones + 1) : 1 << (k - ones));
